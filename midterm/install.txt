import torch
from torch.utils.data import random_split, DataLoader
from torchvision.transforms import ToTensor, Normalize, Compose
from torchvision.datasets import MNIST
import numpy as np

def get_mnist(data_path: str = './data'):

    tr = Compose([ToTensor(), Normalize((0.1307,),(0.3081,))])

    trainset = MNIST(data_path, train=True, download=True, transform=tr)
    testset = MNIST(data_path, train=False, download=True, transform=tr)


    return trainset, testset



import numpy as np

def prepare_dataset(num_partitions: int,
                    batch_size: int,
                    val_ratio: float = 0.1):

    trainset, testset = get_mnist()

    data = trainset.data
    targets = trainset.targets

    num_shards = num_partitions * 2  # client당 2개 shard
    shard_size = len(trainset) // num_shards

    # label 기준 정렬
    idxs = np.arange(len(trainset))
    labels = targets.numpy()

    idxs_labels = np.vstack((idxs, labels))
    idxs_labels = idxs_labels[:, idxs_labels[1].argsort()]
    sorted_idxs = idxs_labels[0]

    # shard 생성
    shards = [sorted_idxs[i * shard_size:(i + 1) * shard_size] for i in range(num_shards)]

    trainloaders = []
    valloaders = []

    for i in range(num_partitions):
        selected_shards = np.random.choice(len(shards), 2, replace=False)

        client_idxs = np.concatenate([shards[j] for j in selected_shards])

        # 선택된 shard 제거
        shards = [shards[k] for k in range(len(shards)) if k not in selected_shards]

        # dataset 생성
        subset = torch.utils.data.Subset(trainset, client_idxs)

        num_total = len(subset)
        num_val = int(val_ratio * num_total)
        num_train = num_total - num_val

        for_train, for_val = random_split(
            subset,
            [num_train, num_val],
            torch.Generator().manual_seed(2023)
        )

        trainloaders.append(DataLoader(for_train, batch_size=batch_size, shuffle=True))
        valloaders.append(DataLoader(for_val, batch_size=batch_size, shuffle=False))

    testloader = DataLoader(testset, batch_size=128)

    return trainloaders, valloaders, testloader
